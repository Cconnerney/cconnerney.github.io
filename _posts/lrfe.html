Logistic Regression¶

lr.py implements a sentiment polarity analyzer using binary logistic regression. The model learns the parameters of a binary logistic regression model and predicts a sentiment polarity (i.e. label) for the corresponding feature vector of each movie review. The program outputs the labels of the training and test examples and calculates training and test error (percentage of incorrectly labeled reviews). The small dataset and large dataset output files reference the output after 500 training epochs.

The model is defined as follows:

import sys
import numpy as np
import math
import csv

formatted_train_input = sys.argv[1]
formatted_validation_input = sys.argv[2]
formatted_test_input = sys.argv[3]
dict_input = sys.argv[4]
train_out = sys.argv[5]
test_out = sys.argv[6]
metrics_out = sys.argv[7]
num_epoch = int(sys.argv[8])

def openfile(input):
    features = []
    with open(input) as myfile:
        for line in myfile:
            temp = line.split()
            features.append(temp)

    data = np.array(features,dtype=float)
    data = np.insert(data,1,1,axis=1)
    params = np.zeros(len(data[0])-1)
    return data,params

def update(xi,params,learnrate,num):
    dot = np.dot(xi[1:],params)
    l1 = (xi[0] -(math.exp(dot)/(1+math.exp(dot))))
    g = ((xi[1:]/num)*l1)
    update = np.add(params,learnrate*g)
    return update

def obj(data,params):
    sum = 0
    for xi in data:
        dot = np.dot(xi[1:],params)
        if xi[0]== 1:
            calc = (math.log(math.exp(dot)/(1+math.exp(dot))))
            sum += calc
        elif xi[0]== 0:
            calc = (math.log((1-(math.exp(dot)/(1+math.exp(dot))))))
            sum += calc
    result = sum/(-(len(data)))
    return result

def train(epochs,data,params,learnrate):
    num = len(data)
    for i in range(epochs):
        for l in range(len(data)):
            params = update(data[l],params,learnrate,num)
        #ans2 =  obj(data,params)
        #print("J(0):",ans2)
    return params

def predict(data,params):
    result = []
    count = 0
    for i in range(len(data)):
        new = data[count,1:]
        dot = np.dot(new,params)
        sig = math.exp(dot)/(1+math.exp(dot))
        if sig >= 0.5:
            out = 1
        else:
            out = 0
        result.append(out)
        count += 1
        if count == len(data):
            count = 0
    return result

def erate(data,result):

    error_list = zip(data,result)
    
    wrong_count = 0
    for item in error_list:
        if item[0] != item[1]:
            wrong_count +=1
    
    error_rate = float(wrong_count) / len(result)
    error_rate = "{:.6f}".format(error_rate)

    return error_rate

def main():

    learnrate = 0.01

    # ----- training
    data,params = openfile(formatted_train_input)
    train_params = train(num_epoch,data,params,learnrate)
    result = predict(data,train_params)

    error_rate = erate(data[:,0],result)

    with open(train_out,'w') as outfile:
        np.savetxt(outfile,result,delimiter='\n',fmt='%d')

    # ----- validation
    data,params = openfile(formatted_validation_input)
    result = predict(data,train_params)

    error_rate1 = erate(data[:,0],result)

    # ----- testing
    data,params = openfile(formatted_test_input)
    result = predict(data,train_params)

    error_rate2 = erate(data[:,0],result)

    with open(test_out,'w') as outfile:
        np.savetxt(outfile,result,delimiter='\n',fmt='%d')

    # ------ metrics

    metric_output = []
    metric_output.append('error(train): '+str(error_rate)+'\n')
    metric_output.append('error(test): '+str(error_rate2))

    with open(metrics_out,'w') as outfile:
        outfile.writelines(metric_output)

        

if __name__ == '__main__':
    main()

Feature Engineering for Logistic Regression¶

feature.py, converts raw data into formatted training, validation and test data. Model 1 is based on the vocabulary information in the dictionary file; Model 2 takes in an additional feature dictionary file. This program transfers the whole movie review text into a feature vector using feature extraction methods.

import sys
import csv
import numpy as np

train_input = sys.argv[1]
validation_input = sys.argv[2]
test_input = sys.argv[3]
dict_input = sys.argv[4]
formatted_train_out = sys.argv[5]
formatted_validation_out = sys.argv[6]
formatted_test_out = sys.argv[7]
feature_flag = int(sys.argv[8])
feature_dictionary_input = sys.argv[9]

def bagofwords(input_file, dict_input, output_file):
    
    features = []
    with open(input_file) as myfile:
        for line in myfile:
            temp = line.split()
            features.append(temp)

    dict1 = []
    with open(dict_input) as myfile:
        for line in myfile:
            temp = line.split()
            dict1.append(temp)
    fdict = np.array(dict1)    

    fdict = dict(zip(fdict[:,0],fdict[:,1]))
    out = np.zeros([len(features),len(dict1)],dtype=int)

    for n,line in enumerate(features):
        for word in line:
            if word in fdict:
                out[n,int(fdict[word])]=1

    labels = [int(i[0]) for i in features]
    labels = np.array(labels).reshape(len(labels),1)
    out = np.hstack((labels,out))

    with open(output_file,'w') as outfile:
        tsv_writer = csv.writer(outfile,delimiter='\t')
        for line in out:
            tsv_writer.writerow(line)

def word2vec(input_file, feature_dictionary_input, output_file):

    features = []
    with open(input_file) as myfile:
        for line in myfile:
            temp = line.split()
            features.append(temp)

    fvec = []
    with open(feature_dictionary_input) as myfile:
        for line in myfile:
            temp = line.split()
            fvec.append(temp)
    
    vdict = {}
    for line in fvec:
        vdict[line[0]]= line[1:]

    newar = np.empty([0,301],dtype='float')
    for line in features:
        new = np.zeros(300)
        count = 0
        for word in line[1:]:
            if word in vdict:
                array = np.array(vdict[word],dtype='float')
                new = np.add(new,array)
                count+=1
        results = new/count
        newar = np.vstack((newar,(np.insert(results,0,line[0],axis=0))))

    with open(output_file,'w') as outfile:
        np.savetxt(outfile,newar,delimiter='\t',fmt='%.6f')

def main():
    
    if feature_flag == 1:
        bagofwords(train_input, dict_input, formatted_train_out)
        bagofwords(validation_input, dict_input, formatted_validation_out)
        bagofwords(test_input, dict_input, formatted_test_out)
    elif feature_flag == 2:
        word2vec(train_input, feature_dictionary_input, formatted_train_out)
        word2vec(validation_input, feature_dictionary_input, formatted_validation_out)
        word2vec(test_input, feature_dictionary_input, formatted_test_out)

    '''
    Full data
    model 1: 1m
    model 2: 35s
    '''

if __name__ == '__main__':
    main()
