Unsupervised Learning¶

Exploring the K-means algorithm in the context of face modeling. The dataset is from Faces in the Wild, a collection of recognized pictures of faces (http://vis-www.cs.umass.edu/lfw/). A subset of the images have been cropped, resized, and turned into black and white.

import gzip
from collections import Counter
import numpy as np
import matplotlib
import matplotlib.pyplot as plt

global DATA_CACHE
DATA_CACHE = None

def get_faces_data():
    global DATA_CACHE
    if DATA_CACHE is None:
        with gzip.open('./faces_all.txt.gz', 'rt') as f:
            data = np.loadtxt(f)
        with gzip.open('./labels_all.txt.gz') as f:
            labels = [line.rstrip() for line in f]
        DATA_CACHE = (data, labels)
    return DATA_CACHE

faces = get_faces_data()[0]

def plot_face_images(data, rows, cols):

    fig,ax = plt.subplots(rows,cols)

    pics = data[:rows*cols,:]
    picdata = []
    for n,row in enumerate(pics):
        row = row.reshape((60,60))
        picdata.append(row)
    count = 0
    for n in range(rows):
        for n2 in range(cols):
            ax[n,n2].axis('off')
            ax[n,n2].imshow(picdata[count],cmap='gray',vmin=0, vmax=255)
            count+=1

    return fig

fig = plot_face_images(faces, 5, 5)

def kmeans(X, k, center_indices, max_iters):
    """ Run the k-means algorithm max_iters iterations, to generate k clusters
    """
    Mu = X[center_indices]
    loss_list = np.zeros((max_iters))
    for i in range(max_iters):
        D = -2*X@Mu.T + (X**2).sum(axis=1)[:,None] + (Mu**2).sum(axis=1)

        C = np.eye(k)[np.argmin(D,axis=1),:]
        loss = np.linalg.norm(X - Mu[np.argmin(D,axis=1),:])**2
        loss_list[i] = loss

        Mu = (C.T @ X)/np.sum(C,axis=0)[:,None]

    C = np.array(list(np.argmax(row) for row in C))
    loss_list = loss_list/len(X)
    return (Mu, C, loss_list)

centers, indices, objective = kmeans(faces, 5, np.array([ 710, 216,  881, 1003, 480]), 30)
print(centers[0][:20])
print(indices[:20])
print(objective)

def kmeans_pp_nn(X, center_index_0, k, j):
    """ Compute initial cluster points using the (deterministic) K-means++ algorithm
    """

    indices = np.array([center_index_0])
    Mu = X[center_index_0].reshape(len(indices),len(X[0]))
    for i in range(1,k):   

        D = -2*X@Mu.T + (X**2).sum(axis=1)[:,None] + (Mu**2).sum(axis=1)

        d_min = np.min(D,axis=1)
        sorted = np.sort(d_min)[len(D)-j-1]
        newind = np.where(d_min == sorted)[0][0]
        indices = np.append(indices,newind)

        Mu = X[indices].reshape(len(indices),len(X[0]))

    return  list(indices)

centers, indices, objective = kmeans(faces, 20, kmeans_pp_nn(faces, 55, 20, 100), 35)
print(objective)
